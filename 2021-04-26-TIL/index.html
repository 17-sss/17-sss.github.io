  • 오늘 작업한 코드 일부!

    const Carousel = ({
  itemsPerCnt = 4,
  carouselWrapStyle = null,
  transitionProperty = 'all',
  transitionDuration = '0.4s',
  arrowOption: {
    // arrowOption을 정의했지만 일부분을 안했을 때 이 안에 있는 값이 적용됨
    leftIcon = '◀',
    rightIcon = '▶',
    arrowLocation = 'normal',
    btnSize = '32px',
  } = {
    // 아예 arrowOption을 정의 안했을 때
    leftIcon: '◀',
    rightIcon: '▶',
    arrowLocation: 'normal',
    btnSize: '32px',
  },
  children: items,
}) => {
  return <StyledCarousel>{/* 생략... */}</StyledCarousel>;
};
