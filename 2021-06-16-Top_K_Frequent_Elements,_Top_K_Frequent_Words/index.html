    /**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
const topKFrequent = (nums, k) => {
    const map = new Map();
    const nMax = nums.length;

    let nIdx = 0;
    while (nIdx < nMax) {
        map.has(nums[nIdx])
            ? map.set(nums[nIdx], map.get(nums[nIdx]) + 1)
            : map.set(nums[nIdx], 1);
        nIdx++;
    }

    return [...map.entries()]
        .sort((a, b) => b[1] - a[1])
        .map((item) => item[0])
        .slice(0, k);
};
