const partition = (nums = [], left, right) => {
    let i = left;
    let j = right;
    let pivot = nums[left];
    while (i < j) {
        while (nums[j] > pivot) j--;
        while (i < j && nums[i] <= pivot) i++;
        swap(nums, i, j);
    }
    nums[left] = nums[j];
    nums[j] = pivot;
    return j;
};
const quickSort = (nums = [], left, right) => {
    if (left < right) {
        const pivot = partition(nums, left, right);
        quickSort(nums, left, pivot - 1);
        quickSort(nums, pivot + 1, right);
    }
    return nums;
};
